38  virtual std::string get_family(std::string const& name) const override

40  /*
41  * This benefits from some explanation.
42  * Some Cisco platforms have multiple runtime environments.
43  * For these platforms, the name reports as the same regardless of
44  * the environment (e.g., "nexus"), but we want the family to report
45  * appropriately according to the environment (e.g., "cisco-wrlinux"
46  * versus "RedHat").
47  *
48  * In order to achieve this goal, we first check to see what would
49  * be reported if we were a standard Linux environment (e.g., a
50  * Linux distro that detects its name as "centos" would map to
51  * family "RedHat"). Only if a standard Linux family is not
52  * detected do we fall back on the information given in our Cisco
53  * release info file.
54  */

81  virtual std::tuple<std::string, std::string> parse_release(std::string const& name, std::string const& release) const override
82  {
83  /*
84  * Cisco software versions can be idiosyncratic.
85  * NX-OS looks something like '7.0(3)I2(0.455)'
86  * IOS XR looks something like '6.0.0.06I'
87  */
88  auto pos = release.find('.');
89  if (pos != std::string::npos) {
90  auto second = release.find('(', pos + 1);
91  if (second == std::string::npos) {
92  second = release.find('.', pos + 1);
93  }
94  if (second == std::string::npos) {
95  return std::make_tuple(release.substr(0, pos), std::string());
96  }
97  return std::make_tuple(release.substr(0, pos),
98  release.substr(pos + 1, second - (pos + 1)));
99  }
100  return std::make_tuple(release, std::string());
